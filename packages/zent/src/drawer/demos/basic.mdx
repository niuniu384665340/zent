```js
import { useState } from 'react';
import { Button, Drawer } from 'zent';

function SimpleDrawer() {
  const [visible, setVisible] = useState(false);
  return (
    <>
      <Button onClick={() => setVisible(true)}>显示</Button>
      <Drawer
        title="Drawer Title"
        footer={
          <div style={{ 'text-align': 'center' }}>
            <Button onClick={() => setVisible(false)}>关闭</Button>
          </div>
        }
        visible={visible}
        onClose={() => setVisible(false)}
      >
        <Button style={{ margin: '10px 20px' }} onClick={() => setVisible(false)}>
          关闭
        </Button>
        {new Array(30).fill().map((_, index) => (
          <div style={{ 'padding': '10px' }}>drawer Content{index}</div>
        ))}
      </Drawer>
    </>
  );
}

ReactDOM.render(<SimpleDrawer />, mountNode);
```
